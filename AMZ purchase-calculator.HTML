function calculateTip() { let price = parseFloat(document.querySelector('#price').value); let discount = parseFloat(document.querySelector('#discount').value) || 0; let taxRate = parseFloat(document.querySelector('#taxRate').value); if (isNaN(price) || isNaN(taxRate)) { alert("Please enter valid values"); return; } let netPrice = price - discount; let tax = netPrice * (taxRate / 100); let total = netPrice + tax; document.querySelector('.tip').style.display = 'block'; document.querySelector('#total').innerHTML = `Net Price: $${netPrice.toFixed(2)}
Tax: $${tax.toFixed(2)}
Total: $${total.toFixed(2)}`; }
